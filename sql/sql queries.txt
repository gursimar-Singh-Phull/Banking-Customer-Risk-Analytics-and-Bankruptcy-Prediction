/* creating tables  */

CREATE TABLE Customers (
    Customer_ID INT PRIMARY KEY,
    Age INT,
    Gender VARCHAR(20),
    Region VARCHAR(50),
    Employment_Status VARCHAR(50),
    Annual_Income DECIMAL(15,2),
    Dependents INT,
    Monthly_Income DECIMAL(15,2)
);

CREATE TABLE Accounts (
    Account_ID INT PRIMARY KEY,
    Customer_ID INT,
    Account_Type VARCHAR(50),
    Account_Balance DECIMAL(15,2),
    Account_Status VARCHAR(20),
    Avg_Transaction DECIMAL(15,2),
    Total_Amount DECIMAL(15,2)
);

CREATE TABLE Transactions (
    Transaction_ID INT PRIMARY KEY,
    Account_ID INT,
    Transaction_Type VARCHAR(50),
    Amount DECIMAL(15,2),
    Merchant_Category VARCHAR(100),
    Channel VARCHAR(50)
);

CREATE TABLE Loans (
    Loan_ID INT PRIMARY KEY,
    Customer_ID INT,
    Loan_Type VARCHAR(50),
    Loan_Amount BIGINT,
    Outstanding_Debt BIGINT,
    Interest_Rate DECIMAL(5,2),
    Monthly_EMI INT,
    Missed_Payments INT
);

/* Classify customers into High, Medium, and Low debt risk categories */

WITH Debt_Risk AS
(
    SELECT
        c.Customer_ID,
        c.Monthly_Income,
        l.Outstanding_Debt,
        l.Monthly_EMI,
        l.Missed_Payments
    FROM Customers c
    JOIN Loans l
    ON c.Customer_ID = l.Customer_ID
)
SELECT *,
CASE
    WHEN Missed_Payments >= 5
         OR Outstanding_Debt > Monthly_Income * 24
    THEN 'HIGH'
    WHEN Missed_Payments BETWEEN 2 AND 4
    THEN 'MEDIUM'
    ELSE 'LOW'
END AS Debt_Trap_Risk
FROM Debt_Risk;


/* Identify customers with the highest outstanding debt and missed payments */

SELECT
    Customer_ID,
    Outstanding_Debt,
    Missed_Payments
FROM Loans
ORDER BY Outstanding_Debt DESC, Missed_Payments DESC
LIMIT 20;


/* Analyze contribution of each loan type to total debt */

SELECT
    Loan_Type,
    COUNT(*) AS Total_Customers,
    SUM(Outstanding_Debt) AS Total_Debt,
    AVG(Outstanding_Debt) AS Avg_Debt
FROM Loans
GROUP BY Loan_Type
ORDER BY Total_Debt DESC;


/* Compare debt levels across different regions */

SELECT
    c.Region,
    COUNT(*) AS Total_Customers,
    SUM(l.Outstanding_Debt) AS Total_Debt,
    AVG(l.Outstanding_Debt) AS Avg_Debt
FROM Customers c
JOIN Loans l
ON c.Customer_ID = l.Customer_ID
GROUP BY c.Region
ORDER BY Total_Debt DESC;


/* Evaluate performance of different account types */

SELECT
    Account_Type,
    COUNT(*) AS Total_Accounts,
    AVG(Account_Balance) AS Avg_Balance,
    SUM(Account_Balance) AS Total_Balance
FROM Accounts
GROUP BY Account_Type
ORDER BY Total_Balance DESC;


/* Analyze distribution of transaction types */

SELECT
    Transaction_Type,
    COUNT(*) AS Transaction_Count,
    ROUND(
        COUNT(*) * 100.0 /
        (SELECT COUNT(*) FROM Transactions),
        2
    ) AS Percentage
FROM Transactions
GROUP BY Transaction_Type
ORDER BY Percentage DESC;


/* Identify the most frequently used transaction channels */

SELECT
    Channel,
    COUNT(*) AS Total_Transactions
FROM Transactions
GROUP BY Channel
ORDER BY Total_Transactions DESC;


/* Determine top spending categories based on transaction amount */

SELECT
    Merchant_Category,
    COUNT(*) AS Number_Transactions,
    SUM(Amount) AS Total_Spent,
    AVG(Amount) AS Avg_Spent
FROM Transactions
GROUP BY Merchant_Category
ORDER BY Total_Spent DESC;


/* Find customers with the largest loan amounts */

SELECT
    Customer_ID,
    Loan_Amount,
    Outstanding_Debt
FROM Loans
ORDER BY Loan_Amount DESC
LIMIT 20;


/* Compare missed payment behavior across loan types */

SELECT
    Loan_Type,
    AVG(Missed_Payments) AS Avg_Missed_Payments,
    SUM(Missed_Payments) AS Total_Missed_Payments
FROM Loans
GROUP BY Loan_Type
ORDER BY Avg_Missed_Payments DESC;


/* Calculate debt-to-income ratio for customer risk assessment */

SELECT
    c.Customer_ID,
    c.Annual_Income,
    l.Outstanding_Debt,
    ROUND(
        l.Outstanding_Debt /
        NULLIF(c.Annual_Income,0),
        2
    ) AS Debt_Income_Ratio
FROM Customers c
JOIN Loans l
ON c.Customer_ID = l.Customer_ID
ORDER BY Debt_Income_Ratio DESC;


/* Identify customers with the highest account balances */

SELECT
    Customer_ID,
    Account_Balance
FROM Accounts
ORDER BY Account_Balance DESC
LIMIT 20;


/* Create a consolidated customer financial summary */

SELECT
    c.Customer_ID,
    c.Region,
    c.Annual_Income,
    a.Account_Balance,
    l.Loan_Amount,
    l.Outstanding_Debt,
    l.Missed_Payments
FROM Customers c
LEFT JOIN Accounts a
ON c.Customer_ID = a.Customer_ID
LEFT JOIN Loans l
ON c.Customer_ID = l.Customer_ID
LIMIT 100;
